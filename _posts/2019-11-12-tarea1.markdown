{

    def muestras(muestra,d,w):
    F = np.random.uniform(-1000,1000,(muestra,d))
    y = []
    for i in range(len(F)):
        x = np.r_[1,F[i]]
        y.append(np.sign(np.dot(w,x)))
    return np.c_[F,y]
    
    muestra = 100
    F = muestras(muestra, d, w)
